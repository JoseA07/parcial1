--7 Traiga el nombre del barrio y el número de colegios ubicados en cada barrio de aquellas instituciones ubicadas en la comuna de buenos aires ordenado por el número de colegios de mayor a menor.
	 --Columnas: barrio, numero_colegios
SELECT colegios.barrio,
COUNT (*) OVER (ORDER BY barrio) AS numero_colegios
FROM COMUNAS
INNER JOIN colegios
ON COMUNAS.ID = colegios.comuna_id
WHERE COMUNAS.NOMBRE = 'BUENOS AIRES'
GROUP BY colegios.barrio
ORDER BY numero_colegios DESC;

-- Traiga los registros junto con el nombre de su comuna, para cada registro deberá calcularse el total de los estudiantes según los contadores. También deberá traer el total de estudiantes agrupados por comuna.
	 --Columnas: ID, NOMBRE_SEDE, COMUNA_ID, NOMBRE_COMUNA, TOTAL_GENERAL, TOTAL_POR COMUNA

SELECT COMUNAS.ID, colegios.NOMBRE_SEDE, colegios.COMUNA_ID,
SUM(contador_prejardin_jardin + contador_transicion + contador_primaria + contador_secundaria + contador_media + 	contador_adultos) AS TOTAL_GENERAL
FROM COMUNAS
INNER JOIN colegios
ON COMUNAS.ID = colegios.comuna_id
GROUP BY COMUNAS.ID, colegios.NOMBRE_SEDE, colegios.COMUNA_ID
ORDER BY TOTAL_POR_COMUNA;


--7.4 Traiga el nombre de los rectores y el número de colegios encargados para cada rector de aquellos rectores que estén encargados de más de 2 colegios. 
	--Los registros deberán estar ordenados alfabéticamente
--	Nota: NO se deberán incluir aquellos registros que tengan un correo electrónico registrado y tampoco aquellos que registran como nombre "s/d"
	--Columnas: rector, numero_colegios
  SELECT rector,
  COUNT(*) OVER (PARTITION BY rector) AS numero_colegios
  FROM colegios
  WHERE --numero_colegios > 2
   rector LIKE '@'
  AND rector = "s/d"
  GROUP BY rector
  order by numero_colegios;
